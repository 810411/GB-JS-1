var a = 1, b = 1, c, d;

c = ++a; alert(c); // 2
- значение a преинкрементируется (1 + 1), и присваивается в переменную c;

d = b++; alert(d); // 1
- в переменную d присваивается значение b (1), затем b постинкрементируется, выводится d = 1;

c = (2+ ++a); alert(c); // 5
- значение a после первого примера равно 2, преинкрементируется на 1 (2 + 1) результат записывается в переменную a
и затем происходит операция сложения 2 + 3, и присвоение результата в c;
(инкремент как унарный оператор имеет больший приоритет над сложением и присвоением);

d = (2+ b++); alert(d); // 4
- значение b после второго примера равно 2, постинкрементируется на 1 (2 + 1) но в отличииии от примера с переменной a
в операцию сложения попадет значение переменной a до инкремента, происходит операция сложения 2 + 2,
и присвоение резултата в c;

alert(a); // 3
alert(b); // 3
в результате предыдущих операций a и b по два раза инкрементировались (эквивалентно вычилению a = 1, a = a(1) + 1 (=2),
a = a(2) + 1 (=3)).
